    function handlePlatformChange(selectElement) {
        const platformLogo = document.getElementById('platformLogo');
        const autresField = document.getElementById('autresField');
        const selectedOption = selectElement.options[selectElement.selectedIndex];
        const logoSrc = selectedOption.getAttribute('data-logo');

        if (logoSrc) {
            platformLogo.src = logoSrc;
            platformLogo.classList.remove('opacity-0');
            platformLogo.classList.add('opacity-100');
        } else {
            platformLogo.classList.remove('opacity-100');
            platformLogo.classList.add('opacity-0');
            platformLogo.src = ""; // Clear the src if no logo
        }

        if (selectedOption.value === 'Autres') {
            autresField.classList.remove('hidden');
        } else {
            autresField.classList.add('hidden');
        }
    }
</script>
